* Crear la app con Vite
  # npm create vite@latest          // Crear la app react con la ultima version de vite
  # npm create vite@2.9.5           // Crear la app react con una version especifica de vite
    ? Project name: cotizador
    ? Select a framework: react
    ? Select a variant: react
  # cd cotizador
  # npm install
  # npm run dev

* Instalar TailwindCSS
  # npm install tailwindcss postcss autoprefixer
  # npm tailwindcss init            // Crea los archivos de configuracion de tailwindcss
  # npx tailwindcss init -p         // Crea los archivos de configuracion de tailwindcss y postcss
